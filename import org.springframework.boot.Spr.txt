import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.bind.annotation.*;

@SpringBootApplication
@RestController
public class SHLWebApp {

    public static void main(String[] args) {
        SpringApplication.run(SHLWebApp.class, args);
    }

    // Health check (IMPORTANT for evaluator)
    @GetMapping("/health")
    public String health() {
        return "Application is running";
    }

    // Recommendation API
    @PostMapping("/recommend")
    public String recommend(@RequestBody String query) {
        return """
        Query: %s

        Recommended Assessments:
        - Java Developer Assessment
        - Backend Engineering Test
        - OOP Fundamentals Assessment

        Reason:
        These match the required skills and experience level.
        """.formatted(query);
    }
}
